017package net.ljcomputing.sr.service.impl;
018
019import net.ljcomputing.core.exception.NoEntityFoundException;
020import net.ljcomputing.core.exception.RequiredValueException;
021import net.ljcomputing.core.service.AbstractDomainEntityServiceImpl;
022import net.ljcomputing.core.util.DateUtils;
023import net.ljcomputing.sr.domain.Activity;
024import net.ljcomputing.sr.domain.Event;
025import net.ljcomputing.sr.entity.ActivityEntity;
026import net.ljcomputing.sr.entity.EventEntity;
027import net.ljcomputing.sr.repository.ActivityEntityRepository;
028import net.ljcomputing.sr.repository.EventEntityRepository;
029import net.ljcomputing.sr.service.EventDomainEntityService;
030
031import org.springframework.beans.factory.annotation.Autowired;
032import org.springframework.stereotype.Service;
033
034import java.time.LocalDate;
035import java.time.LocalDateTime;
036import java.time.LocalTime;
037import java.time.ZoneId;
038import java.util.Date;
039import java.util.List;
040
041import javax.transaction.Transactional;
042
043/**
044 * Event domain entity service.
045 * 
046 * @author James G. Willmore
047 *
048 */
049@Service
050@Transactional
051public class EventDomainEntityServiceImpl extends
052    AbstractDomainEntityServiceImpl<Event, EventEntity, String, EventEntityRepository>
053    implements EventDomainEntityService {
054
055  /** The activity repository. */
056  @Autowired
057  private ActivityEntityRepository activityRepository;
058
059  /**
060   * @see net.ljcomputing.sr.service.EventDomainEntityService#findByActivity(net.ljcomputing.sr.domain.Activity)
061   */
062  public List<Event> findByActivity(Activity activity)
063      throws RequiredValueException, NoEntityFoundException {
064    if (null == activity) {
065      throw new RequiredValueException(
066          "Cannot search for events without an activity.");
067    }
068
069    return findByActivity(activity.getUuid());
070  }
071
072  /**
073   * @see net.ljcomputing.sr.service.EventDomainEntityService#findByActivity(java.lang.String)
074   */
075  public List<Event> findByActivity(String activityUuid)
076      throws RequiredValueException, NoEntityFoundException {
077    if (null == activityUuid) {
078      throw new RequiredValueException(
079          "Cannot search for events without an activity.");
080    }
081
082    ActivityEntity entity = activityRepository.findOne(activityUuid);
083
084    if (null == entity) {
085      throw new NoEntityFoundException(
086          "Activity provided not found - cannot search for events without an activity.");
087    }
088
089    return strategy.entityToDomain(repository.findByActivity(entity));
090  }
091  
092  /**
093   * @see net.ljcomputing.sr.service.EventDomainEntityService#findEventsForToday()
094   */
095  public List<Event> findEventsForToday() 
096      throws RequiredValueException, NoEntityFoundException {
097    Date start = DateUtils.midnight(new Date());
098    Date end = DateUtils.endOfDay(start);
099    
100    return strategy.entityToDomain(repository.findByStartTimeBetween(start, end));
101  }
102
103}
