017package net.ljcomputing.sr.domain;
018
019import net.ljcomputing.core.domain.Domain;
020import net.ljcomputing.core.entity.AbstractDomain;
021import net.ljcomputing.core.exception.RequiredValueException;
022
023import java.util.ArrayList;
024import java.util.List;
025
026/**
027 * Class defining a work breakdown structure.
028 * 
029 * @author James G. Willmore
030 *
031 */
032public class WorkBreakdownStructure extends AbstractDomain implements Domain {
033
034  /** The Constant serialVersionUID. */
035  private static final long serialVersionUID = -2834895408835732844L;
036
037  /** The name. */
038  private String name;
039
040  /** The description. */
041  public String description;
042
043  /** The activities. */
044  private List<Activity> activities;
045
046  /**
047   * Instantiates a new work breakdown structure.
048   */
049  WorkBreakdownStructure() {
050  }
051
052  /**
053   * Instantiates a new work breakdown structure.
054   *
055   * @param name the name
056   * @throws RequiredValueException the required value exception
057   */
058  public WorkBreakdownStructure(final String name)
059      throws RequiredValueException {
060    if (null == name) {
061      throw new RequiredValueException("name is required.");
062    }
063
064    this.name = name;
065  }
066
067  /**
068   * Gets the name.
069   *
070   * @return the name
071   */
072  public String getName() {
073    return name;
074  }
075
076  /**
077   * Gets the activities.
078   *
079   * @return the activities
080   */
081  public List<Activity> getActivities() {
082    if(null == activities) {
083      activities = new ArrayList<Activity>();
084    }
085    
086    return activities;
087  }
088
089  /**
090   * Adds the activity.
091   *
092   * @param activity the activity
093   */
094  public void addActivity(final Activity activity) {
095    Activity result = getActivityByName(activity.getName());
096
097    if (null == result) {
098      getActivities().add(activity);
099    } else {
100      getActivities().remove(result);
101      getActivities().add(activity);
102    }
103  }
104
105  /**
106   * Gets the activity by name. Return the activity found, or null if the 
107   * activity is not found.
108   *
109   * @param name the name
110   * @return the activity by name
111   */
112  public Activity getActivityByName(final String name) {
113    Activity result = null;
114
115    for (Activity activity : getActivities()) {
116      if (name.equals(activity.getName())) {
117        result = activity;
118        break;
119      }
120    }
121
122    return result;
123  }
124
125  /**
126   * Gets the events by activity.
127   *
128   * @param activity the activity
129   * @return the events by activity
130   */
131  public List<Event> getEventsByActivity(final Activity activity) {
132    Activity found = getActivityByName(activity.getName());
133    List<Event> result = new ArrayList<Event>();
134
135    if (null != found) {
136      result.addAll(found.getEvents());
137    }
138
139    return result;
140  }
141
142  /**
143   * Gets the events by ticket number.
144   *
145   * @param ticketNumber the ticket number
146   * @return the events by ticket number
147   */
148  public List<Event> getEventsByTicketNumber(final String ticketNumber) {
149    List<Event> result = new ArrayList<Event>();
150
151    for (Activity activity : getActivities()) {
152      if (null != activity.getEvents()) {
153        for (Event event : activity.getEvents()) {
154          if (ticketNumber.equals(event.ticketNumber)) {
155            result.add(event);
156          }
157        }
158      }
159    }
160
161    return result;
162  }
163
164  /**
165   * @see net.ljcomputing.core.entity.AbstractDomain#toString()
166   */
167  @Override
168  public String toString() {
169    return "WorkBreakdownStructure [name=" + name + ", description="
170        + description + ", activities=" + activities + ", getUuid()="
171        + getUuid() + "]";
172  }
173}
