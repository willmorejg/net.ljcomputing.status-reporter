017package net.ljcomputing.sr.entity;
018
019import net.ljcomputing.core.domain.Domain;
020import net.ljcomputing.core.entity.AbstractPersistedEntity;
021import net.ljcomputing.core.entity.PersistedEntity;
022import net.ljcomputing.gson.annotation.ExcludeFromJson;
023
024import java.util.List;
025
026import javax.persistence.Column;
027import javax.persistence.Entity;
028import javax.persistence.FetchType;
029import javax.persistence.JoinColumn;
030import javax.persistence.ManyToOne;
031import javax.persistence.OneToMany;
032import javax.persistence.Table;
033import javax.validation.constraints.NotNull;
034
035import org.hibernate.annotations.Fetch;
036import org.hibernate.annotations.FetchMode;
037
038/**
039 * Activity entity.
040 */
041@Entity
042@Table(name = "activity")
043// @Cacheable
044public class ActivityEntity extends AbstractPersistedEntity
045    implements PersistedEntity, Domain {
046
047  /** The Constant serialVersionUID. */
048  private static final long serialVersionUID = 2274287008476627493L;
049
050  /** The description. */
051  private String description;
052
053  /** The name. */
054  private String name;
055
056  /** The wbs. */
057  @ExcludeFromJson
058  private WbsEntity wbs;
059
060  /** The events. */
061  private List<EventEntity> events;
062
063  /**
064   * Gets the description.
065   *
066   * @return the description
067   */
068  @Column(name = "description")
069  public String getDescription() {
070    return description;
071  }
072
073  /**
074   * Sets the description.
075   *
076   * @param description the new description
077   */
078  public void setDescription(String description) {
079    this.description = description;
080  }
081
082  /**
083   * Gets the name.
084   *
085   * @return the name
086   */
087  @Column(name = "name", insertable = true, updatable = false, nullable = false,
088      unique = true)
089  @NotNull(message = "name may not be null")
090  public String getName() {
091    return name;
092  }
093
094  /**
095   * Sets the name.
096   *
097   * @param name the new name
098   */
099  public void setName(String name) {
100    this.name = name;
101  }
102
103  /**
104   * Gets the wbs.
105   *
106   * @return the wbs
107   */
108  @ManyToOne(fetch = FetchType.LAZY, optional = false)
109  @JoinColumn(name = "wbs_uuid", referencedColumnName = "uuid")
110  public WbsEntity getWbs() {
111    return wbs;
112  }
113
114  /**
115   * Sets the wbs.
116   *
117   * @param wbs the new wbs
118   */
119  public void setWbs(WbsEntity wbs) {
120    this.wbs = wbs;
121  }
122
123  /**
124   * Gets the events.
125   *
126   * @return the events
127   */
128  @OneToMany(mappedBy = "activity", fetch = FetchType.EAGER)
129  @Fetch(FetchMode.JOIN)
130  public List<EventEntity> getEvents() {
131    return events;
132  }
133
134  /**
135   * Sets the events.
136   *
137   * @param events the new events
138   */
139  public void setEvents(List<EventEntity> events) {
140    this.events = events;
141  }
142
143  /*
144   * (non-Javadoc)
145   * 
146   * @see net.ljcomputing.core.entity.AbstractDomain#toString()
147   */
148  @Override
149  public String toString() {
150    return "ActivityEntity [description=" + description + ", name=" + name
151        + ", events=" + events + ", getCreatedTs()=" + getCreatedTs()
152        + ", getModifiedTs()=" + getModifiedTs() + ", getUuid()=" + getUuid()
153        + "]";
154  }
155}
